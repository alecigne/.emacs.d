* =git-gutter=

https://github.com/syohex/emacs-git-gutter

#+begin_src emacs-lisp :tangle yes
  (use-package git-gutter
    :hook (prog-mode)
    :delight
    :config
    (global-git-gutter-mode)
    (setq git-gutter:hide-gutter t))
#+end_src
